Enum

Enums, short for enumerations, are a way to create named constant values in C. They allow you to define a set of related named integer constants, making the code more readable and maintainable. Here's how enums work in C: 


            enum enumName {
                constant1,
                constant2,
                // Additional constants...
            };

enum: This keyword is used to define an enumeration type.

enumName: This is the name of the enumeration type.

constant1, constant2, etc.: These are the named constant values of the enumeration. Each constant is assigned an integer value automatically, starting from 0 for the first constant and incrementing by 1 for each subsequent constant. However, you can explicitly assign integer values to constants if desired.

#include <stdio.h>

// Define an enumeration named "Weekday" with named constants for days of the week
enum Weekday {
    SUNDAY,
    MONDAY,
    TUESDAY,
    WEDNESDAY,
    THURSDAY,
    FRIDAY,
    SATURDAY
};

int main() {
    // Declare variables of type "Weekday"
    enum Weekday today, tomorrow;

    // Assign values to variables
    today = MONDAY;
    tomorrow = TUESDAY;

    // Print the values of variables
    printf("Today is %d\n", today);
    printf("Tomorrow is %d\n", tomorrow);

    return 0;
}


#include <stdio.h>

enum Level {
  LOW = 25,
  MEDIUM = 50,
  HIGH = 75
};

int main() {
    enum Level myLevel = LOW;
    printf("My level is %d\n", myLevel);

    return 0;
}


#include <stdio.h>

enum Age {
  TEN = 10,
  ELEVEN,   // Now 11
  TWELVE,   // Now 12
  THIRTEEN  // Now 13
};

int main() {
    enum Age myAge, sisterAge, brotherAge;

    myAge = ELEVEN;
    sisterAge = THIRTEEN;
    brotherAge = TEN;

    printf("I a %d years old\n", myAge);
    printf("My sister is %d years old\n", sisterAge);
    printf("My brother is %d years old\n", brotherAge);

    return 0;
}
